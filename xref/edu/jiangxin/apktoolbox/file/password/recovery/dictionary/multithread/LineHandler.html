7   import org.apache.logging.log4j.Logger;

9   import java.util.concurrent.atomic.AtomicBoolean;
10  
11  public class LineHandler {
12  
13      private static final Logger logger = LogManager.getLogger(LineHandler.class.getSimpleName());
14  
15      private AtomicBoolean success;
16  
17      private Synchronizer synchronizer;
18      private FileChecker fileChecker;
19  
20      private CompleteCallback completeCallback;
21  
22      public LineHandler(FileChecker fileChecker, AtomicBoolean success, Synchronizer synchronizer, CompleteCallback completeCallback) {
23          this.fileChecker = fileChecker;
24          this.success = success;
25          this.synchronizer = synchronizer;
26          this.completeCallback = completeCallback;
27      }
28  
29      public AtomicBoolean getSuccess() {
30          return this.success;
31      }
32  
33      public void handle(String line, long currentLineCount, BigFileReader bigFileReader) {
34          if (success.compareAndSet(true, true) || synchronizer.getCurrentState() != State.WORKING) {
35              return;
36          }
37          synchronizer.setCurrentPassword(line);
38          synchronizer.setProgressBarValue(Math.toIntExact(currentLineCount));
39  
40          if (fileChecker.checkPassword(line)) {
41              if (success.compareAndSet(false, true)) {
42                  logger.info("find password: " + line);
43                  completeCallback.onComplete(line);
44              }
45          } else {
46              if (!success.compareAndSet(true, true) && synchronizer.getCurrentState() == State.WORKING) {
47                  logger.info("try password[" + line + "] failed");
48              }
49          }
50      }
51  
52  }
