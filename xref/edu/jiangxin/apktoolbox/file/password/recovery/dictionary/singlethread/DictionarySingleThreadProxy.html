14  public class DictionarySingleThreadProxy {
15      private static final Logger logger = LogManager.getLogger(DictionarySingleThreadProxy.class.getSimpleName());
16  
17      private boolean isCancelled;
18  
19      private static class DictionarySingleThreadProxyHolder {
20          private static final DictionarySingleThreadProxy instance = new DictionarySingleThreadProxy();
21      }
22  
23      private DictionarySingleThreadProxy() {
24      }
25  
26      public static DictionarySingleThreadProxy getInstance() {
27          return DictionarySingleThreadProxy.DictionarySingleThreadProxyHolder.instance;
28      }
29  
30      public String startAndGet(File file, IChecker checker, String charsSet, Synchronizer synchronizer) {
31          isCancelled = false;
32          Predicate<String> isRecoveringPredicate = password -> (!isCancelled);
33          Function<String, Stream<String>> generator = password -> {
34              synchronizer.setCurrentPassword(password);
35              synchronizer.increaseProgressBarValue();
36              return Stream.of(password.toLowerCase(), password.toUpperCase());
37          };
38          Predicate<String> verifier = checker::checkPassword;
39          try (BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(file), charsSet))) {
40              // java.util.stream.BaseStream.parallel can not increase the speed in test
41              Optional<String> password = br.lines().takeWhile(isRecoveringPredicate).flatMap(generator).filter(verifier).findFirst();
42              return password.orElse(null);
43          } catch (FileNotFoundException e) {
44              logger.info("FileNotFoundException");
45          } catch (IOException e) {
46              logger.info("IOException");
47          }
48          return null;
49      }
50  
51      public void cancel() {
52          isCancelled = true;
53      }
54  }
