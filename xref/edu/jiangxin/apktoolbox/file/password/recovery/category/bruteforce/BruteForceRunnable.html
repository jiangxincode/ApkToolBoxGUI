3   import edu.jiangxin.apktoolbox.file.password.recovery.RecoveryPanel;
4   import edu.jiangxin.apktoolbox.file.password.recovery.checker.IChecker;
5   import edu.jiangxin.apktoolbox.file.password.recovery.exception.UnsupportedVersionException;
6   import org.apache.logging.log4j.LogManager;
7   import org.apache.logging.log4j.Logger;
8   
9   public class BruteForceRunnable implements Runnable {
10      private static final Logger logger = LogManager.getLogger(BruteForceRunnable.class.getSimpleName());
11      private final int taskId;
12      private final BruteForceFuture bruteForceFuture;
13      private final BruteForceTaskParam consts;
14  
15      RecoveryPanel panel;
16  
17      public BruteForceRunnable(int taskId, BruteForceTaskParam consts, BruteForceFuture bruteForceFuture, RecoveryPanel panel) {
18          this.taskId = taskId;
19          this.consts = consts;
20          this.bruteForceFuture = bruteForceFuture;
21          this.panel = panel;
22      }
23  
24      @Override
25      public void run() {
26          long rangeBegin = taskId * consts.getPasswordSubRangeSize();
27          long rangeEnd = (taskId + 1) * consts.getPasswordSubRangeSize() - 1;
28          logger.info("rangeBegin: {}, rangeEnd: {}", rangeBegin, rangeEnd);
29          String passwordOrNull = findPasswordInRange(rangeBegin, rangeEnd, consts.getChecker());
30          bruteForceFuture.set(passwordOrNull);
31      }
32  
33      /*
34       * The findPasswordInRange method find the original password using md5 hash function
35       * if a thread discovers the password, it returns original password string; otherwise, it returns null;
36       */
37      public String findPasswordInRange(long rangeBegin, long rangeEnd, IChecker checker) {
38          int[] passwordIterator = new int[consts.getPasswordLength()];
39          transformDecToBaseN(rangeBegin, passwordIterator, consts.getCharsSet().length());
40          for (long iterator = rangeBegin; iterator <= rangeEnd; iterator++) {
41              if (bruteForceFuture.isDone()) {
42                  logger.info("isDone: ThreadName: {}", Thread.currentThread().getName());
43                  return null;
44              }
45              if (bruteForceFuture.isCancelled()) {
46                  logger.info("isCancelled: ThreadName: {}", Thread.currentThread().getName());
47                  return null;
48              }
49              String password = transformIntToStr(passwordIterator, consts.getCharsSet());
50              int result;
51              try {
52                  result = checker.checkPassword(password) ? 1 : 0;
53              } catch (UnsupportedVersionException e) {
54                  logger.error("UnsupportedVersionException, stop");
55                  result = -1;
56              } catch (Exception e) {
57                  logger.error("Exception, stop", e);
58                  result = -1;
59              } finally {
60                  panel.setCurrentPassword(password);
61                  panel.increaseProgressBarValue();
62              }
63              if (result == 0) {
64                  getNextCandidate(passwordIterator, consts.getCharsSet().length());
65              } else if (result == 1) {
66                  return password;
67              } else {
68                  return null;
69              }
70          }
71          return null;
72      }
73  
74      /*
75       * The transformDecToBaseN transforms decimal into numArray that is base N number system
76       * If you don't understand, refer to the homework01 overview
77       */
78      protected void transformDecToBaseN(long numInDec, int[] numArrayInBaseN, final int charSetLength) {
79          for (int index = consts.getPasswordLength() - 1; index >= 0; index--) {
80              numArrayInBaseN[index] = (int) (numInDec % charSetLength);
81              numInDec = numInDec / charSetLength;
82          }
83      }
84  
85      /*
86       * The getNextCandidate update the possible password represented by N base system
87       */
88      private void getNextCandidate(int[] candidateChars, final int charsSetLength) {
89          int reminder = 1;
90          for (int index = consts.getPasswordLength() - 1; index >= 0; index--) {
91              candidateChars[index] += reminder;
92              reminder = candidateChars[index] / charsSetLength;
93              candidateChars[index] %= charsSetLength;
94          }
95      }
96  
97      /*
98       * We assume that each character can be represented to a number : 0 (0) , 1 (1), 2 (2) ... a (10), b (11), c (12), ... x (33), y (34), z (35)
99       * The transformIntToStr transforms int-array into string (numbers and lower-case alphabets)
100      * int array is password represented by base-N system
101      * return : password String
102      *
103      * For example, if you write the code like this,
104      *     int[] pwdBase36 = {10, 11, 12, 13, 0, 1, 9, 2};
105      *     String password = transfromIntoStr(pwdBase36);
106      *     System.out.println(password);
107      *     output is abcd0192.
108      *
109      */
110     private static String transformIntToStr(int[] charsArray, final String charsSet) {
111         char[] password = new char[charsArray.length];
112         for (int i = 0; i < password.length; i++) {
113             password[i] = charsSet.charAt(charsArray[i]);
114         }
115         return new String(password);
116     }
