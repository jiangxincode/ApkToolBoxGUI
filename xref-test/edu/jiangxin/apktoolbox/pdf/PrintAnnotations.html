23      String path = "target/test-classes/pdf/";
24      @Test
25      public void testPrintAnnotations() throws IOException {
26          File file = new File(path, "R常用函数整理.pdf");
27          PDDocument doc = Loader.loadPDF(new RandomAccessReadBufferedFile(file));
28          PDPageTree pdPageTree = doc.getDocumentCatalog().getPages();
29          for (PDPage page : pdPageTree) {
30              List<PDAnnotation> pdAnnotations = page.getAnnotations();
31              for (PDAnnotation pdAnnotation : pdAnnotations) {
32                  if (pdAnnotation.getSubtype().equals("Highlight")) {
33                      PDFTextStripperByArea stripperByArea = new PDFTextStripperByArea();
34                      COSDictionary dictionary = pdAnnotation.getCOSObject();
35                      COSArray quadsArray = (COSArray) dictionary.getDictionaryObject(COSName.getPDFName("QuadPoints"));
36                      String str = null;
37                      for (int j = 1, k = 0; j <= (quadsArray.size() / 8); j++) {
38                          COSFloat ULX = (COSFloat) quadsArray.get(0 + k);
39                          COSFloat ULY = (COSFloat) quadsArray.get(1 + k);
40                          COSFloat URX = (COSFloat) quadsArray.get(2 + k);
41                          COSFloat URY = (COSFloat) quadsArray.get(3 + k);
42                          COSFloat LLX = (COSFloat) quadsArray.get(4 + k);
43                          COSFloat LLY = (COSFloat) quadsArray.get(5 + k);
44                          COSFloat LRX = (COSFloat) quadsArray.get(6 + k);
45                          COSFloat LRY = (COSFloat) quadsArray.get(7 + k);
46  
47                          k += 8;
48  
49                          float ulx = ULX.floatValue() - 1;                           // upper left x.
50                          float uly = ULY.floatValue();                               // upper left y.
51                          float width = URX.floatValue() - LLX.floatValue();          // calculated by upperRightX - lowerLeftX.
52                          float height = URY.floatValue() - LLY.floatValue();         // calculated by upperRightY - lowerLeftY.
53  
54                          PDRectangle pageSize = page.getMediaBox();
55                          uly = pageSize.getHeight() - uly;
56  
57                          Rectangle2D.Float rectangle_2 = new Rectangle2D.Float(ulx, uly, width, height);
58                          stripperByArea.addRegion("highlightedRegion", rectangle_2);
59                          stripperByArea.extractRegions(page);
60                          String highlightedText = stripperByArea.getTextForRegion("highlightedRegion");
61  
62                          if (j > 1) {
63                              str = str.concat(highlightedText);
64                          } else {
65                              str = highlightedText;
66                          }
67                      }
68                      System.out.println(str);
69                  }
70              }
