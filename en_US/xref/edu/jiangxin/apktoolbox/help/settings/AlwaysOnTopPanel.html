26      private void createOptionPanel() {
27          optionPanel = new JPanel();

29  
30          JLabel typeLabel = new JLabel("Always on top:");
31          JCheckBox alwaysOnTopCheckBox = new JCheckBox();
32          alwaysOnTopCheckBox.setSelected(conf.getBoolean("always.on.top", false));
33          alwaysOnTopCheckBox.addActionListener(e -> {
34              conf.setProperty("always.on.top", alwaysOnTopCheckBox.isSelected());
35              getFrame().setAlwaysOnTop(alwaysOnTopCheckBox.isSelected());
36              SwingUtilities.updateComponentTreeUI(getFrame());
37              getFrame().refreshSizeAndLocation();
38          });
39  
40          optionPanel.add(typeLabel);
41          optionPanel.add(Box.createHorizontalStrut(Constants.DEFAULT_X_BORDER));
42          optionPanel.add(alwaysOnTopCheckBox);
