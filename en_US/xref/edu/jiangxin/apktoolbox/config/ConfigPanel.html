6   import javax.swing.*;
7   import java.awt.*;
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import java.io.File;
11  import java.io.IOException;
12  import java.net.URI;
13  import java.net.URISyntaxException;
14  
15  public class ConfigPanel extends EasyPanel {
16  
17      private static final long serialVersionUID = 1L;
18  
19  
20      public ConfigPanel() throws HeadlessException {
21          super();
22      }
23  
24      @Override
25      public void initUI() {
26          BoxLayout boxLayout = new BoxLayout(this, BoxLayout.Y_AXIS);
27          setLayout(boxLayout);
28  
29          createPathPanel("Path of 7ZIP(e.g.\"C:/Program Files/7-Zip/7z.exe\")", "https://www.7-zip.org/", Constants.SEVEN_ZIP_PATH_KEY);
30          add(Box.createVerticalStrut(Constants.DEFAULT_Y_BORDER));
31  
32          createPathPanel("Path of RAR(e.g.\"C:/Program Files/WinRAR/Rar.exe\")", "https://www.win-rar.com/", Constants.RAR_PATH_KEY);
33          add(Box.createVerticalStrut(Constants.DEFAULT_Y_BORDER));
34  
35          createPathPanel("Path of RAR(e.g.\"C:/Program Files/WinRAR/WinRAR.exe\")", "https://www.win-rar.com/", Constants.WIN_RAR_PATH_KEY);
36          add(Box.createVerticalStrut(Constants.DEFAULT_Y_BORDER));
37      }
38  
39      private void createPathPanel(String label, String website, String confKey) {
40          JPanel pathPanel = new JPanel();
41          pathPanel.setLayout(new BoxLayout(pathPanel, BoxLayout.Y_AXIS));
42          add(pathPanel);
43  
44          JPanel firstLinePanel = new JPanel();
45          firstLinePanel.setLayout(new BoxLayout(firstLinePanel, BoxLayout.X_AXIS));
46  
47          JLabel pathLabel = new JLabel(label);
48  
49          JButton visitWebsiteButton = new JButton(bundle.getString("download.button"));
50  
51          firstLinePanel.add(pathLabel);
52          firstLinePanel.add(Box.createHorizontalStrut(Constants.DEFAULT_X_BORDER));
53          firstLinePanel.add(visitWebsiteButton);
54          firstLinePanel.add(Box.createHorizontalGlue());
55  
56          JPanel secondLinePanel = new JPanel();
57          secondLinePanel.setLayout(new BoxLayout(secondLinePanel, BoxLayout.X_AXIS));
58  
59          JTextField pathTextField = new JTextField();
60          pathTextField.setText(conf.getString(confKey));
61  
62          JButton pathButton = new JButton(bundle.getString("choose.file.button"));
63  
64          secondLinePanel.add(pathTextField);
65          secondLinePanel.add(Box.createHorizontalStrut(Constants.DEFAULT_X_BORDER));
66          secondLinePanel.add(pathButton);
67  
68          pathPanel.add(firstLinePanel);
69          pathPanel.add(Box.createVerticalStrut(Constants.DEFAULT_Y_BORDER));
70          pathPanel.add(secondLinePanel);
71  
72  
73          visitWebsiteButton.addActionListener(new ActionListener() {
74              @Override
75              public void actionPerformed(ActionEvent e) {
76                  URI uri;
77                  try {
78                      uri = new URI(website);
79                      Desktop.getDesktop().browse(uri);
80                  } catch (URISyntaxException ex) {
81                      logger.error("URISyntaxException", ex);
82                  } catch (IOException ex) {
83                      logger.error("IOException", ex);
84                  }
85              }
86          });
87  
88          pathButton.addActionListener(new ActionListener() {
89              @Override
90              public void actionPerformed(ActionEvent e) {
91                  JFileChooser jfc = new JFileChooser();
92                  jfc.setFileSelectionMode(JFileChooser.FILES_ONLY);
93                  jfc.setDialogTitle("select a file");
94                  int ret = jfc.showDialog(new JLabel(), null);
95                  switch (ret) {
96                      case JFileChooser.APPROVE_OPTION:
97                          File file = jfc.getSelectedFile();
98                          pathTextField.setText(file.getAbsolutePath());
99                          conf.setProperty(confKey, pathTextField.getText());
100                         break;
101                     default:
102                         break;
103                 }
104             }
105         });
106     }
107 }
