350     private void execute() {
351         String combinedOptionCmd = getCombinedOptionCmd();
352 
353         boolean isOk = checkCondition();
354         if (!isOk) {
355             return;
356         }
357 
358         String logLevel = queryLogLevelString();
359         String logFile = textLogPath.getText() + "\\" + LOG_NAME + DateUtils.getCurrentDateString() + ".txt";
360 
361         monkeyCmd = new String[] { CMD, "/C",
362                 "adb -s " + comboBoxDevices.getSelectedItem() + " shell monkey -p "
363                         + comboBoxProgram.getSelectedItem() + combinedOptionCmd + " --throttle "
364                         + textMillisecond.getText() + "" + logLevel + "" + TRANSACTIONCOUNT + " > " + logFile };
365 
366         logger.info("Monkey: {}", monkeyCmd[2]);
367 
368         flag = 0;
369 
370         try {
371             monkeyProcess = Runtime.getRuntime().exec(monkeyCmd);
372         } catch (Exception ex) {
373             logger.error("Exception", ex);
374         }
375 
376         executeButton.setEnabled(false);
377         resetButton.setEnabled(false);
378         interruptButton.setEnabled(true);
379 
380         // 倒计时文本
381         labelHour.setVisible(true);
382         labelMinute.setVisible(true);
383         labelSecond.setVisible(true);
384 
385         // 获取时间类型
386         String timeType = (String) comboBoxTime.getSelectedItem();
387         long time = Long.parseLong(textTime.getText());
388         if (Objects.equals(timeType, HOUR)) {
389             time *= 3600;
390         } else if (Objects.equals(timeType, MINUTE)) {
391             time *= 60;
392         }
393         threadTimeType = new Thread(new CountdownRunnable(time));
394         threadTimeType.start();
395     }

508     private List<String> getDevices() {
509         List<String> devices = new ArrayList<>();
510         logger.info("get device list start");
511         Process process = null;
512         try {
513             ProcessBuilder builder = new ProcessBuilder("adb", "devices");
514             process = builder.start();
515         } catch (IOException e) {
516             logger.error("exec command failed: " + e.getMessage());
517         }
518         if (process == null) {
519             logger.error("process is null");
520             return devices;
521         }
522         try (BufferedReader reader = new BufferedReader(new InputStreamReader(process.getInputStream(), StandardCharsets.UTF_8))) {
523             String line;
524             while ((line = reader.readLine()) != null) {
525                 if (line.contains("device") && !line.contains("List of")) {
526                     line = line.substring(0, line.length() - 7);
527                     logger.info("device name: " + line);
528                     devices.add(line);
529                 }
530             }
531         } catch (IOException e) {
532             logger.error("read failed: " + e.getMessage());
533         } finally {
534             IOUtils.closeQuietly(process.getOutputStream());
535             IOUtils.closeQuietly(process.getErrorStream());
536             IOUtils.closeQuietly(process.getInputStream());
537         }
538         logger.info("get device list end");
539         return devices;
540     }

542     private List<String> getApplication(String device) {
543         List<String> apps = new ArrayList<>();
544         logger.info("get application list start");
545         Process process = null;
546         try {
547             process = Runtime.getRuntime().exec(new String[]{"adb", "-s", device, "shell", "pm", "list", "packages"});
548         } catch (IOException e) {
549             logger.error("exec command failed: " + e.getMessage());
550         }
551         if (process == null) {
552             logger.error("process is null");
553             return apps;
554         }
555         try (BufferedReader reader = new BufferedReader(new InputStreamReader(process.getInputStream(), StandardCharsets.UTF_8))) {
556             String line;
557             while ((line = reader.readLine()) != null) {
558                 if (line.contains("package:")) {
559                     logger.info("application name：" + line);
560                     apps.add(line.replace("package:", ""));
561                 }
562             }
563         } catch (IOException e) {
564             logger.error("read failed: " + e.getMessage());
565         } finally {
566             IOUtils.closeQuietly(process.getOutputStream());
567             IOUtils.closeQuietly(process.getErrorStream());
568             IOUtils.closeQuietly(process.getInputStream());
569         }
570         logger.info("get application list end");
571         return apps;
572     }

574     private void executeCommand(String[] cmd, String keyValue) {
575         logger.info("exec cmd start");
576         list = new ArrayList<>();
577         Process process = null;
578         try {
579             process = Runtime.getRuntime().exec(cmd);
580         } catch (IOException e) {
581             logger.error("exec command failed: " + e.getMessage());
582         }
583         if (process == null) {
584             logger.error("process is null");
585             return;
586         }
587         try (BufferedReader reader = new BufferedReader(new InputStreamReader(process.getInputStream(), StandardCharsets.UTF_8))) {
588             String line;
589             while ((line = reader.readLine()) != null) {
590                 if (line.contains(keyValue)) {
591                     line = line.replace(" ", ":");
592                     System.out.println("[" + keyValue + "][" + line + "]");
593                     String[] str = line.split(":");
594                     System.out.println(str.length);
595                     for (int i = 1; i < str.length; i++) {
596                         if (str[i].length() > 0) {
597                             logger.info("Pid: " + str[i]);
598                             list.add(str[i]);
599                             break;
600                         }
601                     }
602                 }
603             }
604         } catch (IOException e) {
605             logger.error("read failed: " + e.getMessage());
606         } finally {
607             IOUtils.closeQuietly(process.getOutputStream());
608             IOUtils.closeQuietly(process.getErrorStream());
609             IOUtils.closeQuietly(process.getInputStream());
610         }
611         logger.info("exec cmd end");
612     }

614     /**
615      * 倒计时
616      * 
617      */
618     class CountdownRunnable implements Runnable {
619         private long times;
620 
621         public CountdownRunnable(long times) {
622             this.times = times;
623         }
624 
625         @Override
626         public void run() {
627             // 自定义倒计时时间
628             long time = times;
629             long hour = 0;
630             long minute = 0;
631             long seconds = 0;
632 
633             if (time >= 0) {
634                 for (int i = 0; i <= time; i++) {
635                     // 监控Monkey命令
636                     monitorMonkey(time, MONKEY);
637                     // 监控Monkey命令操作的App
638                     monitorApp(time, comboBoxProgram.getSelectedItem().toString());
639                     // 判断标识，是否中断操作
640                     System.out.println("flag==1?Interrupt:Continue:" + flag);
641                     if (flag == 1) {
642                         break;
643                     }
644                     hour = time / 3600;
645                     minute = (time - hour * 3600) / 60;
646                     seconds = time - hour * 3600 - minute * 60;
647                     labelHour.setText(hour + HOUR);
648                     labelMinute.setText(minute + MINUTE);
649                     labelSecond.setText(seconds + SECOND);
650                     try {
651                         Thread.sleep(1000);
652                     } catch (InterruptedException e) {
653                         logger.error("InterruptedException {}", e.getMessage());
654                         Thread.currentThread().interrupt();
655                     }
656                     // 正常结束
657                     if (time == 0) {
658                         Thread.currentThread().interrupt();
659                         interruptThread();
