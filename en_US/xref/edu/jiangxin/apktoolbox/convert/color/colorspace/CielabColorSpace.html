5   public class CielabColorSpace extends ColorSpace {
6       private static ColorSpace instance = null;
7   
8       private static final long serialVersionUID = 5027741380892134289L;

10      private static final ColorSpace CIEXYZ =
11              getInstance(CS_CIEXYZ);
12  
13      private static final double N = 4.0 / 29.0;
14  
15      CielabColorSpace() {
16          super(TYPE_Lab, 3);
17      }
18  
19      public static synchronized ColorSpace getInstance() {
20          if (instance == null) {
21              instance = new CielabColorSpace();
22          }
23          return instance;
24      }
25  
26      @Override
27      public float[] fromCIEXYZ(float[] colorvalue) {
28          double l = f(colorvalue[1]);
29          double L = 116.0 * l - 16.0;
30          double a = 500.0 * (f(colorvalue[0]) - l);
31          double b = 200.0 * (l - f(colorvalue[2]));
32          return new float[] {(float) L, (float) a, (float) b};
33      }
34  
35      @Override
36      public float[] toCIEXYZ(float[] colorvalue) {
37          double i = (colorvalue[0] + 16.0) * (1.0 / 116.0);
38          double X = fInv(i + colorvalue[1] * (1.0 / 500.0));
39          double Y = fInv(i);
40          double Z = fInv(i - colorvalue[2] * (1.0 / 200.0));
41          return new float[] {(float) X, (float) Y, (float) Z};
42      }
43  
44      @Override
45      public float[] fromRGB(float[] rgbvalue) {
46          float[] xyz = CIEXYZ.fromRGB(rgbvalue);
47          return fromCIEXYZ(xyz);
48      }
49  
50      @Override
51      public float[] toRGB(float[] colorvalue) {
52          float[] xyz = toCIEXYZ(colorvalue);
53          return CIEXYZ.toRGB(xyz);
54      }
55  
56      @Override
57      public float getMaxValue(int component) {
58          return 128f;
59      }
60  
61      @Override
62      public float getMinValue(int component) {
63          return (component == 0)? 0f: -128f;
64      }
65  
66      @Override
67      public String getName(int idx) {
68          return String.valueOf("Lab".charAt(idx));
69      }
70  
71      private static double f(double x) {
72          if (x > 216.0 / 24389.0) {
73              return Math.cbrt(x);
74          } else {
75              return 841.0 / 108.0 * x + N;
76          }
77      }
78  
79      private static double fInv(double x) {
80          if (x > 6.0 / 29.0) {
81              return x*x*x;
82          } else {
83              return 108.0 / 841.0 * (x - N);
84          }
85      }
86  
87      private Object readResolve() {
88          return getInstance();
89      }
