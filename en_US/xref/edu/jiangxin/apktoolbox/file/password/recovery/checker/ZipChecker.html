38      @Override
39      public boolean checkPassword(String password) {
40          if (DEBUG) {
41              logger.info("checkPassword: " + password);
42          }
43          boolean result = false;
44          byte[] readBuffer = new byte[4096];
45          try (ZipInputStream zipInputStream = new ZipInputStream(new FileInputStream(file), password.toCharArray(), StandardCharsets.UTF_8)) {
46              while (zipInputStream.getNextEntry() != null) {
47                  while (zipInputStream.read(readBuffer) != -1) {
48                      // do nothing
49                  }
50              }
51              result = true;
52          } catch (ZipException e) {
53              if (DEBUG) {
54                  logger.error("checkPassword: ", e.getClass());
55              }
56          } catch (IOException e) {
57              throw new UnknownException(e);
58          }
59          return result;
60      }
