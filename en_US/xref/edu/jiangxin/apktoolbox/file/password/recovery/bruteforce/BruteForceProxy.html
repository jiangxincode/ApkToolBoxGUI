12  public class BruteForceProxy {
13      private static final Logger logger = LogManager.getLogger(BruteForceProxy.class.getSimpleName());
14  
15      private ExecutorService executorService;
16  
17      private BruteForceFuture bruteForceFuture;
18  
19      private static class BruteForceProxyHolder {
20          private static final BruteForceProxy instance = new BruteForceProxy();
21      }
22  
23      private BruteForceProxy() {
24      }
25  
26      public static BruteForceProxy getInstance() {
27          return BruteForceProxyHolder.instance;
28      }
29  
30      public String startAndGet(int numThreads, int passwordLength, IChecker checker, String charsSet, Synchronizer synchronizer) {
31          executorService = Executors.newFixedThreadPool(numThreads);
32          bruteForceFuture = new BruteForceFuture(numThreads);
33          BruteForceTaskParam param = new BruteForceTaskParam(numThreads, passwordLength, checker, charsSet);
34          for (int taskId = 0; taskId < numThreads; taskId++) {
35              BruteForceRunnable task = new BruteForceRunnable(taskId, param, bruteForceFuture, synchronizer);
36              executorService.execute(task);
37          }
38          String password = null;
39          try {
40              password = bruteForceFuture.get();
41          } catch (Exception e) {
42              logger.info("Exception test: ", e);
43          } finally {
44              if (executorService.isShutdown()) {
45                  executorService.shutdown();
46              }
47          }
48          return password;
49      }
50  
51      public void cancel() {
52          bruteForceFuture.cancel(false);
53          if (executorService != null && !executorService.isShutdown()) {
54              executorService.shutdown();
55          }
56          while (executorService != null && !executorService.isTerminated()) {
57              try {
58                  final boolean isTimeout = !executorService.awaitTermination(100, TimeUnit.SECONDS);
59                  logger.info("awaitTermination isTimeout: " + isTimeout);
60              } catch (InterruptedException e) {
61                  logger.error("awaitTermination InterruptedException");
62                  Thread.currentThread().interrupt();
63              }
64          }
65      }
66  }
